  5. function uncompress(str) {
    let result = "";
    
    for (let i = 0; i < str.length; i++) {
        let number = "";

        while (!isNaN(str[i + 1])) {
            number += str[i + 1];
            i++;
        } 

        if (str.length == 1) {
            result = str;
        } else {
            result += str[i - number.length].repeat(number);
        }

        number = "";
    }

    console.log(`"${str}" => "${result}"`);
